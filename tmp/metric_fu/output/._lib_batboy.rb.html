4# Main Batboy Stats Reporting App
  
5# This is the app-loading and startup lib file. You should be able to

7class Batboy

10  def initialize(ostream=$stdout, stats_grinder=StatsGrinder.new)

13  end

36  def report_most_improved_batter_in(year)
  
37    from, to = year-1, year
  
38    ostream.puts "Most improved batter #{from}->#{to}:"
  
39    batter = stats_grinder.most_improved_batter(from, to)
  
40    ostream.puts batter.name
  
41    ostream.puts
  
42  end
  
43
  
44  def report_slugging_percentage_roster_for(team, year)
  
45    ostream.puts "#{year} Slugging percentages for #{team}:"
  
46    batters = stats_grinder.team_members_for_year(team, year)
  
47    batters_with_stats = batters.map {|batter|
  
48      [batter, batter.stats_for_year(year).slugging_percentage]
  
49    }
  
50    batters_with_stats.sort_by(&:last).reverse
  
51    batters_with_stats.each do |batter, slugging_percentage|
  
52      ostream.puts "%20s: %6.3f" % [batter.name, slugging_percentage]
  
53    end
  
54    ostream.puts
  
55  end
  
56
  
57  def report_triple_crown_winner_in_league_for(league, year)
  
58    ostream.puts "#{year} #{league} Triple Crown Winner:"
  
59    if winner = stats_grinder.triple_crown_winner_in_league_for(league, year)
  
60      ostream.puts winner.name
  
61    else
  
62      ostream.puts "(No winner)"
  
63    end
  
64  end
  
65end
