5# Representation of a player. Class-level finders search the dataset
  
6# while instance methods provide stats and convenience lookup into
  
7# player demographic data

14  def initialize(id:, last_name:, first_name:)
  
15    @id, @last_name, @first_name = id, last_name, first_name
  
16    @batting_data = Hash.new do |hash, year|
  
17      hash[year] = Hash.new do |hash2, league_id|
  
18        hash2[league_id] = Hash.new do |hash3, team_id|
  
19          hash3[team_id] = BattingData.new player_id: id
  
20        end
  
21      end
  
22    end
  
23  end

40  def self.find_all(year: nil, league: nil, team: nil)
  
41    return batter_data.map(&:last) unless year
  
42    return find_all_by_year(year) unless league || team
  
43    # subtle: if you supply both league and team, ignore league
  
44    return find_all_by_league_and_year(league, year) unless team
  
45    find_all_by_team_and_year(team, year)
  
46  end

74  def played_any_games?(year:, league: nil, team: nil)
  
75    return played_any_games_in_year?(year) unless league || team
  
76    return played_any_games_in_league_in_year?(league, year) unless team
  
77    played_any_games_for_team_in_year?(team, year)
  
78  end

100  private_class_method def self.find_all_by_year(year)
  
101    batter_data.reject {|id, batter|
  
102      !batter.played_any_games?(year: year)
  
103    }.map(&:last)
  
104  end

106  private_class_method def self.find_all_by_team_and_year(team, year)
  
107    batter_data.reject {|id, batter|
  
108      !batter.played_any_games?(year: year, team: team)
  
109    }.map(&:last)
  
110  end

112  private_class_method def self.find_all_by_league_and_year(league, year)
  
113    batter_data.reject {|id, batter|
  
114      !batter.played_any_games?(year: year, league: league)
  
115    }.map(&:last)
  
116  end

197  def played_any_games_in_year?(year)
  
198    years.include? year
  
199  end
  
200
  
201  def played_any_games_for_team_in_year?(team, year)
  
202    played_any_games?(year: year) && batting_data[year].any? {|_, team_data|
  
203      team_data.keys.include?(team)
  
204    }
  
205  end
  
206
  
207  def played_any_games_in_league_in_year?(league, year)
  
208    played_any_games?(year: year) && batting_data[year].any? {|league_name, _|
  
209      league == league_name
  
210    }
  
211  end

215  def all_batting_data_ever
  
216    batting_data.map {|year, league_data|
  
217      all_batting_data_for_year league_data

219  end
