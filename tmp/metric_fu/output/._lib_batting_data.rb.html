4# Stats aggregation/integration class
  
5class BattingData
  
6
  
7  attr_reader :player_id, :player, :year, :league, :team, :games,
  
8              :at_bats, :runs, :hits, :doubles, :triples, :home_runs,
  
 9              :runs_batted_in, :stolen_bases, :caught_stealing
  
10
  
11  private_attr_writer :player_id, :player, :year, :league, :team,
  
12                      :games, :at_bats, :runs, :hits, :doubles,
  
13                      :triples, :home_runs, :runs_batted_in,
  
14                      :stolen_bases, :caught_stealing
  
15
  
16  def initialize(data={})
  
17    @player_id = data.fetch(:player_id)
  
18    @player = Batter.find(id: player_id)
  
19    @year = data[:year].to_i
  
20    @league, @team = data[:league], data[:team]
  
21
  
22    integrable_stats.each {|key| send "#{key}=", data[key].to_i }
  
23  end
  
24
  
25  def integrable_stats
  
26    [
  
27     :games, :at_bats, :runs, :hits, :doubles, :triples, :home_runs,
  
28     :runs_batted_in, :stolen_bases, :caught_stealing
  
29    ]
  
30  end
  
31
  
32  def batting_average
  
33    return 0.0 unless at_bats > 0
  
34    hits / at_bats.to_f
  
35  end
  
36
  
37  def slugging_percentage
  
38    return 0.0 unless at_bats > 0
  
39    bases_advanced / at_bats.to_f
  
40  end
  
41
  
42  # Add BattingData to another and return a new BattingData containing
  
43  # the sums of the stats.
  
44  def +(other)
  
45    # It may seem odd to init player, year, league and team here, but we may
  
46    # be adding a valid BattingData to a player's empty BattingData,
  
47    # and if so we want to return the valid one.
  
48    ors = %i(player_id player year league).each_with_object({}) {|key, hash|
  
49      hash[key] = send(key) || other.send(key)
  
50    }
  
51    sums = integrable_stats.each_with_object({}) {|key, hash|
  
52      hash[key] = send(key) + other.send(key)
  
53    }
  
54
  
55    BattingData.new(ors.merge sums)
  
56  end
  
57
  
58  private
  
59
  
60  def bases_advanced
  
61    singles + 2*doubles + 3*triples + 4*home_runs
  
62  end
  
63
  
64  def singles
  
65    hits - (doubles + triples + home_runs)
  
66  end
  
67end
