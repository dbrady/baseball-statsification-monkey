3# StatsGrinder is a class that is aware of BattingData from a higher
  
4# level than BattingData itself; specifically it is able to compare
  
5# BattingData objects to each other to determine improvement, identify
  
6# triple crown winners, etc. It also knows how to search for relevant
  
7# Batters to satisfy a given query.
  
8#
  
 9# It may be possible to offload some of the stats to BattingData. For
  
10# example, BattingData#batting_average_improvement(other_batting_data)
  
11# but for now the purpose of StatsGrinder is to satisfy our curiosity
  
12# about various baseball statistics that are fussy and hand-tweaked
  
13# (like only considering batters with a certain number of at-bats for
  
14# a given report)
  
15class StatsGrinder
  
16  def most_improved_batter(from, to)
  
17    batters1 = Batter.find_all year: from
  
18    batters1 = with_at_least_200_at_bats(batters1, from)
  
19    batters2 = Batter.find_all year: to
  
20    batters2 = with_at_least_200_at_bats(batters2, to)
  
21
  
22    batters = common_batters(batters1, batters2)
  
23
  
24    most_improved(batters, from, to)
  
25  end
  
26
  
27  # So far Batboy has only ever needed to talk to StatsGrinder. No
  
28  # need for it to talk directly to Batter just yet--let's proxy this
  
29  # for now and get an intention-revealing name in the bargain.
  
30  def team_members_for_year(team, year)
  
31    Batter.find_all(team: team, year: year)
  
32  end
  
33
  
34  def triple_crown_winner_in_league_for(league, year)
  
35    # BattingData already has everything we need here -- just need to
  
36    # teach Batter to cobble up and return BattingDatas that only
  
37    # include data from the desired league and year. Then ask each
  
38    # batter for stats_for_league_and_year, and see if the max_by
  
39    # :home_runs, :runs_batted_in, and :batting_average are all the
  
40    # same person.
  
41    contenders = Batter.find_all league: league, year: year
  
42    contenders = with_at_least_400_at_bats_in_league(contenders, year, league)
  
43
  
44    homer = contenders.max_by {|batter|
  
45      batter.stats_for_league_and_year(league, year).home_runs
  
46    }
  
47    run_batter_inner = contenders.max_by {|batter|
  
48      batter.stats_for_league_and_year(league, year).runs_batted_in
  
49    }
  
50    best_hitter = contenders.max_by {|batter|
  
51      batter.stats_for_league_and_year(league, year).batting_average
  
52    }
  
53    if homer == run_batter_inner && homer == best_hitter
  
54      homer
  
55    else
  
56      nil # ;-)
  
57    end
  
58  end
  
59
  
60  private
  
61
  
62  # Isolating this method because the coding exercise does not
  
63  # adequately explain how to calculate improvement -- do we want
  
64  # percentage improvement, or raw points gained? Fortunately, for
  
65  # the year in question the most improved batter wins by a wide
  
66  # enough margin to take both raw points AND percentage, so here
  
67  # endeth my pondering--it's isolated and I'm done. :-) [Update: this
  
68  # was bugging me so I called my baseball-stats-loving friend, and he
  
69  # informed me that it's preferred to calculate based on raw points,
  
70  # not percentage, because batting average is already a percentage,
  
71  # and grinding out "percentages of percentages" becomes
  
72  # statistically problematic.]
  
73  def batter_improvement(batter, from, to)
  
74    tba, fba = batter.stats_for_year(to).batting_average,
  
75         batter.stats_for_year(from).batting_average
  
76    improvement = tba - fba
  
77    # improvement_percent = improvement / fba
  
78  end
  
79
  
80  def with_at_least_200_at_bats(batters, year)
  
81    batters.reject {|batter| batter.stats_for_year(year).at_bats < 200 }
  
82  end
  
83
  
84  def with_at_least_400_at_bats_in_league(batters, year, league)
  
85    batters.reject {|batter|
  
86      batter.stats_for_league_and_year(league, year).at_bats < 400
  
87    }
  
88  end
  
89
  
90  # reject batters not present in both years
  
91  def common_batters(batters1, batters2)
  
92    batter2_ids = batters2.map(&:id)
  
93    batters1.reject {|batter| !batter2_ids.include?(batter.id) }
  
94  end
  
95
  
96  # return most-improved batter in set for the 2 years in
  
97  # question. This could be extracted into a sort_by / first method
  
98  # pair but ruby already gives us max_by.
  
 99  def most_improved(batters, from, to)
  
100    batters_with_improvements = batters.map {|batter|
  
101      [batter, batter_improvement(batter, from, to)]
  
102    }.max_by {|b, bwi| bwi }.first
  
103  end
  
104end
